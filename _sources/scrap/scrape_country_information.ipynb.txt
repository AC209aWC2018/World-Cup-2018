    "    continents = continents[:-2]\n",
    "    country_to_continent = {}\n",
    "    \n",
    "    for continent in continents:\n",

    "            country_name = country_name.replace('*','')\n",
    "            country_to_continent[country_name] = continent.text\n",
    "    \n",
    "    return country_to_continent"
